11def srs_to_spatialreference(srs: str) -> 'osgeo.osr.SpatialReference': 

12 """Convert coordinates system as string to OSR spatial reference 

13 

14 Args: 

15 srs (str): coordinates system PROJ4 compliant, with authority and code, like EPSG:3857 or IGNF:LAMB93 

16 

17 Raises: 

18 RuntimeError: Provided SRS is invalid for OSR 

19 

20 Returns: 

21 osgeo.osr.SpatialReference: Corresponding OSR spatial reference 

22 """ 

23 

24 authority, code = srs.split(':', 1) 

25 

26 sr = osr.SpatialReference() 

27 if authority.upper() == "EPSG": 

28 sr.ImportFromEPSG(int(code)) 

29 else: 

30 sr.ImportFromProj4(f"+init={srs.upper()} +wktext") 

31 

32 return sr

34def bbox_to_geometry(bbox: Tuple[float, float, float, float], srs: str = None, densification: int = 0) -> 'osgeo.ogr.Geometry': 

35 """Convert bbox coordinates to OGR geometry 

36 

37 Args: 

38 bbox (Tuple[float, float, float, float]): bounding box (xmin, ymin, xmax, ymax) 

39 srs (str, optional): coordinates system. Defaults to None. 

40 densification (int, optional): Number of point to add for each side of bounding box. Defaults to 0. 

41 

42 Raises: 

43 RuntimeError: Provided SRS is invalid for OSR 

44  

45 Returns: 

46 osgeo.ogr.Geometry: Corresponding OGR geometry, with spatial reference if provided 

47 """ 

48 

49 ring = ogr.Geometry(ogr.wkbLinearRing) 

50 

51 if densification > 0: 

52 step_x = (bbox[2] - bbox[0]) / (densification + 1) 

53 step_y = (bbox[3] - bbox[1]) / (densification + 1) 

54 

55 for i in range(densification + 1): 

56 ring.AddPoint(bbox[0] + step_x * i, bbox[1]) 

57 for i in range(densification + 1): 

58 ring.AddPoint(bbox[2], bbox[1] + step_y * i) 

59 for i in range(densification + 1): 

60 ring.AddPoint(bbox[2] - step_x * i, bbox[3]) 

61 for i in range(densification + 1): 

62 ring.AddPoint(bbox[0], bbox[3] - step_y * i) 

63 ring.AddPoint(bbox[0], bbox[1]) 

64 

65 else: 

66 ring.AddPoint(bbox[0], bbox[1]) 

67 ring.AddPoint(bbox[2], bbox[1]) 

68 ring.AddPoint(bbox[2], bbox[3]) 

69 ring.AddPoint(bbox[0], bbox[3]) 

70 ring.AddPoint(bbox[0], bbox[1]) 

71 

72 

73 geom = ogr.Geometry(ogr.wkbPolygon) 

74 geom.AddGeometry(ring) 

75 geom.SetCoordinateDimension(2) 

76 

77 if srs is not None: 

78 geom.AssignSpatialReference(srs_to_spatialreference(srs)) 

79 

80 return geom

84def reproject_bbox(bbox: Tuple[float, float, float, float], srs_src: str, srs_dst: str, densification: int = 5) -> Tuple[float, float, float, float]: 

85 """Return bounding box in other coordinates system  

86 

87 Points are added to be sure output bounding box contains input bounding box 

88 

89 Args: 

90 bbox (Tuple[float, float, float, float]): bounding box (xmin, ymin, xmax, ymax) with source coordinates system 

91 srs_src (str): source coordinates system 

92 srs_dst (str): destination coordinates system 

93 densification (int, optional): Number of point to add for each side of bounding box. Defaults to 5. 

94 

95 Returns: 

96 Tuple[float, float, float, float]: bounding box (xmin, ymin, xmax, ymax) with destination coordinates system 

97 """ 

98 

99 bbox_src = bbox_to_geometry(bbox, srs_src, densification) 

100 sr_geo = srs_to_spatialreference(srs_dst) 

101 

102 bbox_dst = bbox_src.Clone() 

103 os.environ["OGR_ENABLE_PARTIAL_REPROJECTION"] = "YES" 

104 bbox_dst.TransformTo(sr_geo) 

105 

106 env = bbox_dst.GetEnvelope() 

107 return (env[0], env[2], env[1], env[3])
